
规则IB_OM_RULE_INFO 表

  RULE_NO	规则编号
  RULE_TYP_CD	规则类型代码
  HOLI_FLG	节假日标志
  RULE_TRI_POSITION	规则触发位置
  SUIT_CHNL_SCP	适用渠道范围
  SUIT_LPR_SCP	适用法人范围
  SUIT_ORG_SCP	适用机构范围
  SUIT_TX_SCP	适用交易范围
  RULE_COMNT	规则说明
  EFFT_FLG	生效标志
  OPER_TELR_NO	操作柜员号
  OPER_TM	操作时间
  OPER_RSN	操作原因

  OPRTN_COND_NO	运营条件编号
  DICTRY_NM	字典名称
  OPER_SYM_1	运算符号1
  CMPR_VAL	比较值
  OPER_SYM_2	运算符号2
  VALUE2	比较值2
  TRAN_CD	交易码
  COND_DESC	条件描述
  OPER_TELR_NO	操作柜员号
  OPER_TM	操作时间
  OPER_RSN	操作原因
  CMPR_VAL_DATA_DICTRY_FLG	比较值数据字典标志
  PUB_DICTRY_FLG	公共字典标志
  DICTRY_DESC	字典描述


规则条件映射IB_OM_RULECOND_RLT

  RULE_COND_NO	CMPL_MODE_FLG	OPRTN_RULE_NO

    AUTH_LVL_CD	授权级别代码

    AUTH_ORG_TYP_CD	授权机构类型代码
    AUTH_ORG_NO	授权机构号
    AUTH_POST_NO	授权岗位编号
    UGNT_FLG	加急标志
    AUTH_DESC	授权描述

    HOST_AUTH_TYP_CD	主机授权类型代码

    CNTRTN_AUTH_LVL_CD	集中授权级别代码

  FIELD_CMPR	字段比较符
  DATA_ATTR_DESC	数据属性描述




-- 授权查询
SELECT 
RULE_RLT.SUIT_TX_SCP,
RULE_RLT.RULE_NO,
RULE_RLT.RULE_COND_NO,
MODE.MODE_NO,
MODE.AUTH_TYP_CD,
MODE.AUTH_LVL_CD,
MODE.AUTH_DESC,
COND.DICTRY_NM,
TRAN_KEYWORDS.PRIV_DICTRY_NM,
COND.OPER_SYM_1,
COND.CMPR_VAL,
COND.OPER_SYM_2,
COND.VALUE2,
RULE_RLT.EFFT_FLG,
RULE_RLT.CMPL_MODE_FLG,
RULE_RLT.RULE_TYP_CD,
RULE_RLT.HOLI_FLG,
RULE_RLT.RULE_TRI_POSITION,
RULE_RLT.SUIT_CHNL_SCP,
RULE_RLT.SUIT_LPR_SCP,
RULE_RLT.SUIT_ORG_SCP,
RULE_RLT.RULE_COMNT,
RULE_RLT.OPER_TELR_NO,
RULE_RLT.OPER_TM,
RULE_RLT.OPER_RSN,
COND.OPRTN_COND_NO,
COND.TRAN_CD,
COND.COND_DESC,
COND.OPER_TELR_NO,
COND.OPER_TM,
COND.OPER_RSN,
COND.CMPR_VAL_DATA_DICTRY_FLG,
COND.PUB_DICTRY_FLG,
COND.DICTRY_DESC,
RULE_RLT.OPRTN_RULE_NO,
MODE.REMOTE_AUTH_LVL_CD,
MODE.AUTH_ORG_TYP_CD,
MODE.AUTH_ORG_NO,
MODE.AUTH_POST_NO,
MODE.UGNT_FLG,
MODE.HOST_AUTH_FLG,
MODE.HOST_AUTH_TYP_CD,
MODE.CNTRTN_AUTH_CENT_NM,
MODE.CNTRTN_AUTH_LVL_CD,
MODE.REMRK_1,
MODE.APP_NO
FROM
(
SELECT 
  RULE.RULE_NO,
  RULE.RULE_TYP_CD,
  RULE.HOLI_FLG,
  RULE.RULE_TRI_POSITION,
  RULE.SUIT_CHNL_SCP,
  RULE.SUIT_LPR_SCP,
  RULE.SUIT_ORG_SCP,
  RULE.SUIT_TX_SCP,
  RULE.RULE_COMNT,
  RULE.EFFT_FLG,
  RULE.OPER_TELR_NO,
  RULE.OPER_TM,
  RULE.OPER_RSN,
  RLT.RULE_COND_NO,
  RLT.CMPL_MODE_FLG,
  RLT.OPRTN_RULE_NO
FROM IB_OM_RULECOND_RLT RLT,IB_OM_RULE_INFO RULE
WHERE
RULE.RULE_TYP_CD = 'AU' AND
RULE.SUIT_TX_SCP = '00301010' AND
RLT.OPRTN_RULE_NO = RULE.RULE_NO
)
RULE_RLT
LEFT JOIN IB_OM_RULECOND_INFO COND
ON RULE_RLT.RULE_COND_NO = COND.OPRTN_COND_NO
LEFT JOIN IB_OM_AUTHMODE_INFO MODE
ON MODE.MODE_NO = COND.OPRTN_COND_NO
LEFT JOIN TE_PARA_TRANKEYWORDS_INFO TRAN_KEYWORDS
ON TRAN_KEYWORDS.TRAN_CD = RULE_RLT.SUIT_TX_SCP AND TRAN_KEYWORDS.PUB_DICTRY_NM = COND.DICTRY_NM
